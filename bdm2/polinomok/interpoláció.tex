\subsection{Interpoláció}
\subsubsection{Lagrange alappolinomok}
\begin{definition}
	Legyenek $x_0, \dots, x_n$ különböző értékek.
	Ekkor az $i$-edik Lagrange alappolinom
	\begin{equation*}
		L_i = \prod_{\substack{j = 0 \\ i \neq j}}^n \frac{x - x_j}{x_i - x_j}
	\end{equation*}
\end{definition}
