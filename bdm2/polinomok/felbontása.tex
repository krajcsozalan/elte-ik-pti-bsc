\setcounter{definition}{0}
\setcounter{theorem}{0}
\subsection{Polinomok felbontása}
\subsubsection{Algebra alaptétele}
\begin{theorem}
	Legyen $f \in \mathbb{C}[x]$ polinom, hogy $\deg f \geq 1$.
	Ekkor $f$-nek van gyöke.
\end{theorem}
\begin{theorem}
	Legyen $f \in \mathbb{C}[x]$ egy $n$-ed fokú polinom.
	Ekkor $f$ felírható a következő módon
	\begin{equation*}
		f = a_n \cdot (x - x_1) \cdot \dots \cdot (x - x_n)
	\end{equation*}
\end{theorem}
\begin{remark*}
	A tétel $\mathbb{R}[x]$-ben nem igaz.
\end{remark*}
\subsubsection{Irreducibilis polinomok}
\begin{definition}
	Egy $f$ polinom irreducibilis,
	ha nem bontható szorzatra nem-triviális módon, azaz
	\begin{equation*}
		f = g \cdot h \implies \deg f = \deg g \lor \deg f = \deg h
	\end{equation*}
\end{definition}
\subsubsection{Számelmélet alaptétele}
\begin{theorem}
	Legyen $\mathbb{K} \in \{ \mathbb{C, Q, R, Z}_p \}$.
	Ekkor minden $f \in \mathbb{K}[x]$ polinom felírható
	\begin{equation*}
		f = a \cdot f_1 \cdot \dots \cdot f_\ell \quad
		a \in \mathbb{K}, a \neq 0, f_1, \dots, f_\ell \in \mathbb{K}[x]
	\end{equation*}
	alakban, ahol $f_1, \dots, f_\ell$ egy főegyütthatós irreducibilis polinomok
	és a felírás sorrendjétől eltekintve egyértelmű.
\end{theorem}
