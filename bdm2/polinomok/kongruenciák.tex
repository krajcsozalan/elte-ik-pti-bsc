\subsubsection{Fogalma}
\begin{definition}
	Legyen $\mathbb{K} \in \{ \mathbb{C, R, Q, Z}_p \}$ és
	legyen $h \in \mathbb{K}[x]$ egy nem-nulla polinom.
	Ekkor $f, g \in \mathbb{K}[x]$ polinomokra
	\begin{equation*}
		h \mid f - g \implies f \equiv g \mod h
	\end{equation*}
\end{definition}
