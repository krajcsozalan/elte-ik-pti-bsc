\setcounter{theorem}{0}
\subsection{Gyökei}
\subsubsection{Fogalma}
\begin{definition}
	Egy $f$ polinomnak az $x_0$ érték a gyöke,
	ha a megfelel polinomfüggvény ott a $0$ értéket veszi fel.
	\begin{equation*}
		x_0 \overset{f}{\mapsto} f(x_0) = 0
	\end{equation*}
\end{definition}
\subsubsection{Fokok és gyökszám}
\begin{theorem}
	Legyen $\mathbb{K} = \{ \mathbb{C, R, Q, Z}_p \}$.
	Egy $f \in \mathbb{K}[x]$ polinomnak legfeljebb $\deg f$ gyöke lehet.
\end{theorem}
\begin{remark*}
	A tétel $\mathbb{Z}_8[x]$ nem igaz.
\end{remark*}
\begin{proof}
	\begin{gather*}
		\deg f = 0 \implies f = c_0, c_0 \neq 0 \implies
		f\text{-nek nincs gyöke} \\
		\deg f \geq 1 \land f\text{-nek nincs gyöke} \implies
		f\text{-nek nincs gyöke} \\
		\deg f \geq 1 \land f(x_0) = 0 \implies
		\exists q, r \in \mathbb{K}[x]: f = (x - x_0) \cdot q + r \quad
		\deg r < 1
	\end{gather*}
	\begin{multline*}
		f(x_0) = 0 = q(x_0) \cdot (x_0 - x_0) + r \implies r = 0 \implies \\
		f = (x - x_0) \cdot q \implies \deg q = \deg f - 1
	\end{multline*}
	\begin{equation*}
		\exists x_1: x_1 \neq x_0 \land f(x_1) = 0 \implies
		f(x_1) = (x_1 - x_0) \cdot q(x_1) \implies q(x_1) = 0
	\end{equation*}
	Mivel $q$-nak legfeljebb $n - 1$ gyöke lehet,
	így $f$-nek legfeljebb $n$ gyöke lehet.
\end{proof}
\begin{theorem}
	Legyen $\mathbb{K} \in \{ \mathbb{C, R, Q, Z}_p \}$.
	Legyen $f \in \mathbb{K}[x]$ és $x_0 \in \mathbb{K}$ egy gyöke.
	Ekkor $f$ felírható az $f = (x - x_0) \cdot g$ formában
	valamely $g \in \mathbb{K}[x]$ polinommal.
\end{theorem}
\subsubsection{Többszörös gyök}
\begin{definition}
	Egy $f$ polinomnak az $x_0$ érték gyöke, ha $(x - x_0) \mid f$.
	Egy $f$ polinomnak az $x_0$ érték $k$-szoros gyöke,
	ha $(x - x_0)^k \mid f$, de $(x - x_0)^{k + 1} \nmid f$.
	Egy $f$ polinomnak az $x_0$ érték többszörös gyöke,
	ha legalább kétszeres gyöke.
\end{definition}
\subsubsection{Formális derivált}
\begin{definition}
	Polinomokra definiáljuk az $f'$ formális deriváltat a következő módon
	\begin{itemize}
		\item $(x^n)' = n \cdot x^{n - 1}$
		\item $(c \cdot f)' = c \cdot f'$
		\item $(f + g)' = f' + g'$
	\end{itemize}
\end{definition}
\begin{theorem}
	A formális derivált teljesíti a szorzat szabályt.
	\begin{equation*}
		(f \cdot g)' = f' \cdot g + f \cdot g'
	\end{equation*}
\end{theorem}
\begin{proof}
	\begin{equation*}
		f = c_n \cdot x^n + \dots c_0 \quad
		g = d_m \cdot x^m + \dots d_0 \\
	\end{equation*}
	\begin{multline*}
		(f \cdot g)' = \left( (c_n \cdot x^n + \dots + c_0) \cdot
		(d_m \cdot x^m + \dots + d_0) \right)' = \\
		\left( \sum_{i, j} (c_i \cdot d_j \cdot x^{i + j}) \right)' =
		\sum_{i, j} \left( c_i \cdot d_j \cdot x^{i + j} \right)'
	\end{multline*}
	\begin{multline*}
		(c_i \cdot d_j \cdot x^{i + j}) =
		(i + j) \cdot c_i \cdot d_j \cdot x^{i + j - 1} =
		i \cdot c_i \cdot x^{i - 1} \cdot d_j \cdot x^j +
		c_i \cdot x^i \cdot j \cdot d_j \cdot x^{j - 1} = \\
		(c_i \cdot x^i)' \cdot d_j \cdot x^j +
		c_i \cdot x_i \cdot (d_j \cdot x^j)
	\end{multline*}
\end{proof}
