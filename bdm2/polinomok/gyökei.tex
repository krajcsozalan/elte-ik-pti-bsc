\subsection{Gyökei}
\subsubsection{Fogalma}
\begin{definition}
	Egy $f$ polinomnak az $x_0$ érték a gyöke,
	ha a megfelel polinomfüggvény ott a $0$ értéket veszi fel.
	\begin{equation*}
		x_0 \overset{f}{\mapsto} f(x_0) = 0
	\end{equation*}
\end{definition}
