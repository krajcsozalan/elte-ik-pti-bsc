\subsection{Kriptográfiai alkalmazások}
\subsubsection{Diffie-Hellman kulcscsere}
\begin{definition}
	Legyen $p$ prímszám, $g$ generátor modulo $p$.
	Legyenek a titkos kulcsok $a, b \in \mathbb{Z}_{p - 1}$.
	Legyenek a nyilvános kulcsok $g^a \bmod p, g^b \bmod p$.
	Ekkor a közös titkos kulcs
	\begin{equation*}
		(g^a)^b \equiv (g^b)^a \equiv g^{a \cdot b} \mod p
	\end{equation*}
\end{definition}
\subsubsection{RSA titkosítás}
\begin{definition}
	Legyen $p, q$ két prímszám és $n = p \cdot q$.
	Ekkor $\varphi(n) = (p - 1) \cdot (q - 1)$.
	Legyen $e \geq 2: (e, \varphi(n)) = 1$.
	Legyen $d: e \cdot d \equiv 1 \mod \varphi(n)$.
	Ekkor a titkos kulcs $(p, q, d)$ és a nyilvános kulcs $(n, e)$.
	Legyen $m: 1 \leq m < n, (m, n) = 1$ egy üzenet.
	Ekkor a rejtjelezett üzenet $m^e \bmod n$,
	az eredeti üzenet $(m^e)^d \bmod n$.
\end{definition}
\begin{proof}
	\begin{gather*}
		(m, n) = 1 \\
		(m^e)^d \equiv m^{e \cdot d} \equiv m^{1 + k \cdot \varphi(n)} \equiv
		m \cdot (m^{\varphi(n)})^k \equiv m \cdot 1^k \equiv m \mod n
	\end{gather*}
\end{proof}
\subsubsection{Gyors hatványozás}
\begin{theorem}
	Legyenek $n, a, k$ pozitív egészek, $n > 1$. Ekkor
	\begin{gather*}
		k = \sum_{i = 0}^{\ell} \varepsilon_i \cdot 2^i
		\quad \varepsilon_i \in \{ 0, 1 \} \\
		a^k \equiv a^{\sum_{i = 0}^{\ell} \varepsilon_i \cdot 2^i} \equiv
		\prod_{i = 0}^{\ell} \left( a^{\varepsilon_i} \right)^{2^i} \equiv \\
		\equiv \left(\dots \left(\left( a^{\ell_i} \bmod n \right)^2 \cdot
		a^{\ell_{i - 1}} \bmod n \right)^2 \dots \right)^2 \cdot
		a^{\ell_0} \bmod n
	\end{gather*}
\end{theorem}
