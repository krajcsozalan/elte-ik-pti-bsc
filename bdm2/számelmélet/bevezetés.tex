\subsection{Bevezetés}
\subsubsection{Maradékos osztás}
\begin{theorem}
    Tetszőleges $a, b \neq 0$ egész számokhoz
    egyértelműen léteznek $q, r$ egészek, hogy
    \begin{equation*}
        a = b \cdot q + r \quad 0 \leq r < |b|
    \end{equation*}
\end{theorem}
\begin{remark*}
    Jelölése: $r = a \bmod b$
\end{remark*}
\begin{proof}
    \begin{equation*}
        a < b \implies a = b \cdot 0 + a \implies q = 0 \land r = a
    \end{equation*}
    \begin{multline*}
        a \geq b \implies \exists q', r': a - b = b \cdot q' + r' \implies \\
        a = b \cdot (q' + 1) + r' \implies q = q' + 1 \land r = r'
    \end{multline*}
    \begin{gather*}
        a = b \cdot q + r = b \cdot q' + r' \quad (r' \geq r) \implies
        b \cdot (q - q') = r' - r \\
        q \neq q' \implies |q - q'| \geq 1 \implies
		|b \cdot (q - q')| \geq b \implies |r' - r| = r' - r \geq b
        \tag{Ellentmondás, mert $r, r' < |b|$}
    \end{gather*}
\end{proof}
\subsubsection{Legnagyobb közös osztó}
\begin{definition}
    Legyenek $a, b \in \mathbb{Z}$.
    A $d$ nemnegatív egész szám az $a$ és $b$ legnagyobb közös osztója, ha
    \begin{itemize}
        \item $d \mid a$ és $d \mid b$
        \item minden $k \in \mathbb{Z}$ esetén, ha $k \mid a$ és $k \mid b$,
        akkor $k \mid d$
    \end{itemize}
    $a = b = 0$ esetén $d = 0$.
\end{definition}
\begin{remark*}
    Jelölése: $d = (a, b) = \text{lnko}(a, b) = \gcd(a, b)$
\end{remark*}
\begin{theorem}
    Minden $a, b \in \mathbb{Z}$ esetén
    létezik az $(a, b)$ legnagyobb közös osztó.
\end{theorem}
\begin{proof}
    \begin{equation*}
        (a, b) = (0, 0) \implies (a, b) = 0
    \end{equation*}
    \begin{align*}
        (a, b) \neq (0, 0) \implies \\
        a = b \cdot q_1 + r_1 & \quad 0 < r_1 < |b| \\
        b = r_1 \cdot q_2 + r_2 & \quad 0 < r_2 < r_1 \\
        r_1 = r_2 \cdot q_3 + r_3 & \quad 0 < r_3 < r_2 \\
        \dots \\
        r_{n - 2} = r_{n - 1} \cdot q_n + r_n & \quad 0 < r_n < r_{n - 1} \\
        r_{n - 1} = r_n \cdot q_{n + 1} \\
        (a, b) = r_n
    \end{align*}
\end{proof}
\subsubsection{Euklideszi algoritmus}
\begin{definition}
	Legyen $a, b \in \mathbb{Z}$ és $r_{-1} = a, r_0 = b$.
	Ekkor az euklideszi algoritmus $i$-edik ($i \geq 1$) sora
	\begin{equation*}
		r_{i - 2} = r_{i - 1} \cdot q_i + r_i \quad 0 < r_i < r_{i - 1}
	\end{equation*}
\end{definition}
\begin{theorem}
	Az euklideszi algoritmus véges sok lépésben véget ér és
	az utolsó nem-nulla maradék a legnagyobb közös osztó.
\end{theorem}
\begin{proof}
	\begin{gather*}
		|b| > r_1 > r_2 > \dots > r_{n - 1} > r_n \\
		r_n \mid r_{n - 1} \implies
		r_n \mid r_{n - 1} \cdot q_n + r_n = r_{n - 2} \implies
		\dots \implies r_n \mid b \implies r_n \mid a \\
		c \mid a, b \implies
		c \mid b \cdot q_1 + r_1 - q_1 \cdot b = r_1 \implies
		c \mid r_2 \implies \dots \implies c \mid r_n
	\end{gather*}
\end{proof}
\begin{theorem}
	Legyenek $a, b$ egész számok és tegyük fel, hogy $1 < |b| < a$.
	Ekkor az algoritmus legfeljebb $2 \cdot (1 + \log |b|)$ lépésben véget ér.
\end{theorem}
\begin{proof}
	\begin{gather*}
		r_{i - 2} = r_{i - 1} \cdot q_i + r_i \quad (0 < r_i < r_{i - 1}) \\
		r_{i - 2} > r_i \cdot q_i + r_i > 2 \cdot r_i \\
		n \text{ páratlan} \implies |b| > r_1 > 2 \cdot r_3 > 4 \cdot r_5 >
		\dots > 2^\frac{n - 1}{2} \cdot r_n > 2^\frac{n - 1}{2} \\
		n \text{ páros} \implies |b| > r_1 > 2 \cdot r_3 > 4 \cdot r_5 >
		\dots > 2^\frac{n - 2}{2} \cdot r_{n - 1} > 2^\frac{n - 2}{2} \\
		|b| > 2^\frac{n - 2}{2} \implies n < 2 \cdot (1 + \log |b|)
	\end{gather*}
\end{proof}
\subsubsection{Lineáris diofantikus egyenletek}
\begin{theorem}
	Minden $a, b, c$ egész számok esetén
	pontosan akkor léteznek $x, y$ egészek, hogy $x \cdot a + y \cdot b = c$,
	ha $(a, b) \mid c$.
\end{theorem}
\begin{proof}
	\begin{equation*}
		(a, b) \mid a \land (a, b) \mid b \implies
		(a, b) \mid x \cdot a + y \cdot b = c
	\end{equation*}
	\begin{gather*}
		(a, b) \mid c \implies (a, b) \cdot k = c \\
		x' \cdot a + y' \cdot b = (a, b) \implies
		(x' \cdot k) \cdot a + (y' \cdot k) \cdot b = k \cdot (a, b) = c
	\end{gather*}
	\begin{gather*}
		r_{i - 2} = r_{i - 1} \cdot q_i + r_i \\
		x_{-1} = 1, x_0 = 0 \quad i \geq 1 \implies
		x_i = x_{i - 2} - q_i \cdot x_{i - 1} \\
		y_{-1} = 0, y_1 = 1 \quad i \geq 1 \implies
		y_i = y_{i - 2} - q_i \cdot y_{i - 1} \\
	\end{gather*}
	\begin{gather*}
		i = -1 \implies r_{-1} = a = 1 \cdot a + 0 \cdot b \\
		i = 0 \implies r_0 = b = 0 \cdot a + 1 \cdot b \\
		i \geq 1 \implies r_{i - 2} = r_{i - 1} \cdot q_i + r_i \implies
		r_i = r_{i - 2} - r_{i - 1} \cdot q_i \\
		r_i = \left( x_{i - 2} \cdot a + y_{i - 2} \cdot b \right) -
		\left( x_{i - 1} \cdot a + y_{i - 1} \cdot b \right) \cdot q_i \\
		r_i = \left( x_{i - 2} - q_i \cdot x_{i - 1} \right) \cdot a +
		\left( y_{i - 2} - q_i \cdot y_{i - 1} \right) \cdot b \\
		r_i = x_i \cdot a + y_i \cdot b \\
		r_n = x_n \cdot a + y_n \cdot b
	\end{gather*}
\end{proof}
