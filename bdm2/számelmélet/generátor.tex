\subsection{Generátor és diszkrét logaritmus}
\subsubsection{Generátor}
\begin{definition}
	Legyen $p$ prímszám. Ekkor van olyan $1 < g < p$ egész, hogy
	\begin{equation*}
		\{ g^0 \bmod p, g^1 \bmod p, \dots, g^{p - 2} \bmod p\} =
		\mathbb{Z}_p^* = \{ 1, 2, \dots, p - 1 \}
	\end{equation*}
	Ekkor $g$ egy generátor.
\end{definition}
